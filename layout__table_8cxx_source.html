137  void layout_table::update()
138  {
139  int opt_cwidth, opt_cheight, min_cwidth, min_cheight, chints;
140  int opt_width, opt_height;
141  int min_width, min_height;
142  int border_width, border_height;
143  int new_cwidth, new_cheight;
144  int realized_cwidth, realized_cheight;
145  bool is_wrong_size, need_relayout;
146  int pos_x, pos_y, pos_x_offset, pos_y_offset;
147 
148 
149  if (do_not_layout || !container || !container->get_nr_children() || !nr_cols || !nr_rows)
150  return;
151 
152  true_w = w;
153  true_h = h;
154 
155  is_wrong_size = false;
156 
157  // calculate the space needed for the border and inner spacings
158  border_width = spacings.horizontal.element*(nr_cols-1) + spacings.horizontal.border*2;
159  border_height = spacings.vertical.element*(nr_rows-1) + spacings.vertical.border*2;
160 
161  // initialize the column and row tables and calculate
162  // minimum and optimum size
163  initialize_space_tables();
164 
165  int num_relayouts = 100;
166 
167  // rearrange the children as long as relayouting is neccessary
168  do {
169  num_relayouts--;
170  need_relayout = false;
171 
172  // get optimum and minimum sum size
173  get_sizes(columns, nr_cols, &opt_width, &min_width);
174  get_sizes(rows, nr_rows, &opt_height, &min_height);
175 
176  // add border spaces
177  opt_width+=border_width;
178  min_width+=border_width;
179  opt_height+=border_height;
180  min_height+=border_height;
181 
182  // is it possible to fit the table into the given size?
183  if (min_width > true_w) {
184  is_wrong_size = true;
185  true_w = min_width;
186  }
187  if (min_height > true_h) {
188  is_wrong_size = true;
189  true_h = min_height;
190  }
191 
192 
193  // split the space into cells
194  distribute_space(columns, nr_cols, true_w - border_width);
195  distribute_space(rows, nr_rows, true_h - border_height);
196 
197  for (int r=0; r<nr_rows; r++) {
198 
199  for (int c=0; c<nr_cols; c++) {
200  base_ptr cur_child = get_child(c+r*nr_cols);
201 
202  get_child_default_size(cur_child, opt_cwidth, opt_cheight);
203  get_child_minimum_size(cur_child, min_cwidth, min_cheight);
204  chints = get_child_layout_hints(cur_child);
205 
206  // calculate the child size according to the
207  // specific layout hints for a child
208  calculate_child_size(columns[c], opt_cwidth, min_cwidth, chints, LH_HHINTS, new_cwidth);
209  calculate_child_size(rows[r], opt_cheight, min_cheight, chints, LH_VHINTS, new_cheight);
210 
211  // set size and position for this child
212  set_child_size(cur_child, new_cwidth, new_cheight);
213 
214  // test whether the size request could be fulfilled
215  get_child_size(cur_child, realized_cwidth, realized_cheight);
216 
217  // the width could not be set for this element as it returns a greater
218  // value than we tried to set. Update its minimum width, expand our
219  // container width (which will now not fit anymore) and relayout
220  if (realized_cwidth > columns[c].real_size) {
221  if (columns[c].min_size < realized_cwidth)
222  columns[c].min_size = realized_cwidth;
223  if (columns[c].opt_size < columns[c].min_size)
224  columns[c].opt_size = columns[c].min_size;
225 
226  cur_child->set<int>("mw", realized_cwidth);
227  //true_w += realized_cwidth - columns[c].real_size;
228  is_wrong_size = true;
229  need_relayout = true;
230  break;
231  }
232 
233  if (realized_cheight > rows[r].real_size) {
234  if (rows[r].min_size < realized_cheight)
235  rows[r].min_size = realized_cheight;
236  if (rows[r].opt_size < rows[r].min_size)
237  rows[r].opt_size = rows[r].min_size;
238 
239  cur_child->set<int>("mh", realized_cheight);
240  is_wrong_size = true;
241  need_relayout = true;
242  break;
243  }
244  }
245  }
246  } while(need_relayout && num_relayouts>0);
247 
248  if (num_relayouts == 0) {
249  std::cerr<<"layout_table stopped layouting the GUI because the maximum number of layouting iterations"<<std::endl;
250  std::cerr<<"was reached. This is probably a bug in the layouter."<<std::endl;
251  }
252 
253  // elements are resized now. set the position
254  pos_y = spacings.vertical.border;
255 
256  for (int r=0; r<nr_rows; r++) {
257  pos_x = spacings.horizontal.border;
258  for (int c=0; c<nr_cols; c++) {
259  base_ptr cur_child = get_child(c+r*nr_cols);
260 
261  get_child_size(cur_child, opt_cwidth, opt_cheight);
262  chints = get_child_layout_hints(cur_child);
263 
264  calculate_child_pos(columns[c], opt_cwidth, chints, LH_HHINTS, pos_x_offset);
265  calculate_child_pos(rows[r], opt_cheight, chints, LH_VHINTS, pos_y_offset);
266 
267  set_child_position(cur_child, pos_x + pos_x_offset, pos_y + pos_y_offset);
268 
269  pos_x+=columns[c].real_size + spacings.horizontal.element;
270  }
271  pos_y+=rows[r].real_size + spacings.vertical.element;
272  }
273 
274 
275  if (is_wrong_size) {
276  do_not_layout = true;
277 
278  // if a resize was needed then the default values
279  // (minimum size and default size) do not seem
280  // to be correct. Repair them
281  repair_default_values(min_width, min_height,
282  opt_width, opt_height);
283 
284  // set the actual size
285  container->set<int>("w", true_w);
286  container->set<int>("h", true_h);
287 
288  do_not_layout = false;
289 
290  }

365  void layout_table::update_spaces_table(layout_table_cell &element, int opt_size, int min_size, int hints, int hints_shift)
366  {
367  // set the optimum size for this cell
368  if (opt_size > element.opt_size)
369  element.opt_size = opt_size;
370 
371  // set the minimum size for this cell or if an
372  // element cannot be shrunk set the optimum size as minimum
373  if (!(hints & (LH_HSHRINK << hints_shift)))
374  min_size = opt_size;
375  if (min_size > element.min_size)
376  element.min_size = min_size;
377  if (element.opt_size < element.min_size)
378  element.opt_size = element.min_size;
379 
380  // has the column element an element that shall be
381  // filled, expanded or shrinked?
382  if (hints & (LH_HFILL << hints_shift))
383  element.can_fill = true;
384 
385  if (hints & (LH_HSHRINK << hints_shift))
386  element.can_shrink = true;
387 
388  if (hints & (LH_HEXPAND << hints_shift))
389  element.can_expand = true;

412  bool layout_table::distribute_space(layout_table_cell *elements, int nr_elements, int new_size)
413  {
414  if (new_size<0) {
415  new_size = 0;
416  }
417 
418  int opt_size;
419  // calculate the optimum size
420  get_sizes(elements, nr_elements, &opt_size, NULL);
421  for (int i=0; i<nr_elements; i++)
422  elements[i].real_size = elements[i].opt_size;
423 
424  // do we have enough space? if this is the case then first
425  // look for elements that shall fill the space and distribute
426  // the remaining space in the ratio of their old size
427  if (new_size >= opt_size) {
428  int remain_space = new_size - opt_size;
429 
430  // get the sum of all fillable spaces
431  int fill_space = 0;
432  int expand_space = 0;
433  for (int i=0; i<nr_elements; i++) {
434  if (elements[i].can_fill)
435  fill_space+=elements[i].opt_size;
436  if (elements[i].can_expand)
437  expand_space+=elements[i].opt_size;
438  }
439 
440  // do we have something that shall be filled?
441  // if yes then distribute the space and quit
442  if (fill_space>0) {
443  for (int i=0; i<nr_elements; i++)
444  if (elements[i].can_fill)
445  elements[i].real_size+=elements[i].opt_size*remain_space/fill_space;
446 
447  return true;
448  }
449 
450  // nothing to fill but elements to be expanded?
451  // also distribute the space and quit
452  if (expand_space > 0) {
453  for (int i=0; i<nr_elements; i++)
454  if (elements[i].can_expand)
455  elements[i].real_size+=elements[i].opt_size*remain_space/expand_space;
456 
457  return true;
458  }
459 
460  // still here. expand again but only all elements
461  for (int i=0; i<nr_elements; i++)
462  elements[i].real_size+=elements[i].opt_size*remain_space/opt_size;
463  }
464 
465 
466 
467  // the size is not sufficient, so try to shrink elements
468  if (new_size < opt_size) {
469  // calculate the minimum size
470  int min_size;
471  get_sizes(elements, nr_elements, NULL, &min_size);
472 
473  // is it possible to shrink the elements enough?
474  // if not we can stop trying
475  if (min_size>new_size) {
476  return false;
477  }
478 
479  // how much can be shrunk
480  int shrink_size = opt_size - min_size;
481  // how much has to be shrunk
482  int overfl_size = opt_size - new_size;
483 
484  int shrunk = 0;
485  // shrink all elements according to their ratio
486  for (int i=0; i<nr_elements; i++)
487  if (elements[i].can_shrink) {
488  elements[i].real_size-= (elements[i].opt_size - elements[i].min_size)*overfl_size/shrink_size;
489  shrunk+=(elements[i].opt_size - elements[i].min_size)*overfl_size/shrink_size;
490  }
491 
492  // there might be one pixel left due to a division
493  // by an odd number. get it from the first element
494  // that can be shrunk.
495  if (shrunk != overfl_size)
496  for (int i=0; i<nr_elements; i++)
497  if (elements[i].can_shrink && elements[i].real_size > elements[i].min_size) {
498  elements[i].real_size--;
499  break;
500  }
501  }
502 
503  return true;
504  }
