17 bool data_format::set_data_format(const std::string& description)
18 {
19  if (description.empty()) {
20  dimensions.clear();
21  component_format::clear();
22  return true;
23  }
24  last_error = "";
25  size_t p = description.find_first_of(']');
26  if (p == std::string::npos) {
27  last_error = "no <]> found in description";
28  return false;
29  }
30  if (!component_format::set_component_format(description.substr(0,p+1)))
31  return false;
32  std::string sub_descr = description.substr(p+1);
33  std::vector<token> toks;
34  tokenizer(sub_descr).set_sep("():|,").bite_all(toks);
35  if (toks.size() < 3) {
36  last_error = "incomplete format description";
37  return false;
38  }
39  if (to_string(toks.back()) != ")") {
40  last_error = "format description not terminated by <)>";
41  return false;
42  }
43  unsigned i=0;
44  unsigned di = 0;
45  int last_a = 1;
46  if (to_string(toks[i]) == "|") {
47  if (!is_integer(to_string(toks[1]), last_a)) {
48  last_error = "expected integer after <|>";
49  return false;
50  }
51  i += 2;
52  }
53  if (to_string(toks[i]) != "(") {
54  last_error = "expected <(> to define dimensions";
55  return false;
56  }
57  if (++i == toks.size()) {
58  last_error = "incomplete format description";
59  return false;
60  }
61  dimensions.clear();
62  do {
63  int n, ld = di, a = 1;
64  if (!is_integer(to_string(toks[i]),n)) {
65  last_error = "expected integer after <(> or <,>";
66  return false;
67  }
68  ++i;
69  if (i+1 < toks.size()) {
70  if (to_string(toks[i]) == ":") {
71  if (!is_integer(to_string(toks[i+1]),ld)) {
72  last_error = "expected integer after <:>";
73  return false;
74  }
75  i += 2;
76  }
77  }
78  if (i+1 < toks.size()) {
79  if (to_string(toks[i]) == "|") {
80  if (!is_integer(to_string(toks[i+1]),a)) {
81  last_error = "expected integer after <:>";
82  return false;
83  }
84  i += 2;
85  }
86  }
87  dimensions.push_back(dimension_info(n,last_a,ld));
88  last_a = a;
89  if (to_string(toks[i]) == ")")
90  return true;
91  if (to_string(toks[i]) != ",") {
92  last_error = "expected <,> after definition of dimension";
93  return false;
94  }
95  ++i;
96  ++di;
97  } while (true);
98  return true;
99 }
