672 void convert_color_model(const color<T1,RGB>& rgb, color<T2,HLS>& hls) {
673  double mx, mn;
674  convert_color_component(std::max(rgb[0], std::max(rgb[1],rgb[2])), mx);
675  convert_color_component(std::min(rgb[0], std::min(rgb[1],rgb[2])), mn);
676  double LL = (mx+mn)/2;
677  if (mn == mx) {
678  hls[0] = hls[2] = 0;
679  convert_color_component(LL,hls[1]);
680  }
681  else {
682  double DM = mx-mn;
683  double SM = mx+mn;
684  double SS = (LL <= 0.5) ? DM/SM : DM/(2-SM);
685  color<double,RGB> c(rgb);
686  double HH;
687  if (c[0] == mx) HH = (c[1]-c[2])/DM;
688  else if (c[1] == mx) HH = 2+(c[2]-c[0])/DM;
689  else HH = 4+(c[0]-c[1])/DM;
690  if (HH < 0) HH += 6;
691  HH /= 6;
692  convert_color_component(HH,hls[0]);
693  convert_color_component(LL,hls[1]);
694  convert_color_component(SS,hls[2]);
695  }

699 void convert_color_model(const color<T1,HLS>& hls, color<T2,RGB>& rgb) {

702  convert_color_component(hls[1],LL);
703  convert_color_component(hls[2],SS);
704  int I = (int) HH;
705  double F = HH-I;
706  while (I >= 6) I -= 6;
707  double mx = (LL <= 0.5) ? LL*(1+SS) : LL+SS-LL*SS;
708  double mn = 2*LL - mx;
709  double DM = mx -mn;
710  if (SS == 0) {
711  T2 tmp;
712  convert_color_component(LL,tmp);
713  rgb = tmp;
714  }
715  else {

717  case 0 :

719  convert_color_component(mn+F*DM,rgb[1]);

721  break;

724  convert_color_component(mx, rgb[1]);
