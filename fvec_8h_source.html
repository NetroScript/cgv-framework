536 template <typename T, cgv::type::uint32_type N>
537 const fvec<T, N> slerp(const fvec<T, N> &v0, const fvec<T, N> &v1, T t) {
538  T dotv0v1 = dot(v0, v1);
539  // clamp between [-1,1]
540  if (dotv0v1 < -1)
541  dotv0v1 = -1;
542 
543  if (dotv0v1 > 1)
544  dotv0v1 = 1;
545 
546  T theta = acos(dotv0v1) * t;
547  auto v2 = normalize(v1 - (dotv0v1)*v0);
548  return T(cos(theta)) * v0 + T(sin(theta)) * v2;
549 }
