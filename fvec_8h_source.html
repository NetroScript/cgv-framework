517 template <typename T, cgv::type::uint32_type N>
518 const fvec<T, N> slerp(const fvec<T, N> &v0, const fvec<T, N> &v1, T t) {
519  T dotv0v1 = dot(v0, v1);
520  // clamp between [-1,1]
521  if (dotv0v1 < -1)
522  dotv0v1 = -1;
523 
524  if (dotv0v1 > 1)
525  dotv0v1 = 1;
526 
527  T theta = acos(dotv0v1) * t;
528  auto v2 = normalize(v1 - (dotv0v1)*v0);
529  return T(cos(theta)) * v0 + T(sin(theta)) * v2;
530 }
