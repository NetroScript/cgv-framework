57 bool bmp_reader::open(const std::string& file_name, data_format& df, std::vector<data_format>* palette_formats)
58 {
59  if (fp != 0)
60  close();
61  palette.clear();
62 
63  fp = cgv::base::open_data_file(file_name.c_str(), "rb");
64  if (!fp) {
65  last_error = "could not open file: ";
66  last_error += file_name;
67  return false;
68  }
69  unsigned char bmp_header[14];
70  if (fread(bmp_header, 1, 14, fp) != 14) {
71  last_error = "could not read bmp header"; fclose(fp); fp = 0; return false;
72  }
73  if (GET_2B(bmp_header,0) != 0x4D42) {
74  last_error = "bmp header does not start with BM"; fclose(fp); fp = 0; return false;
75  }
76  unsigned char bmp_info[64];
77  if (fread(bmp_info, 1, 4, fp) != 4) {
78  last_error = "could not read first 4 bytes of bmp info"; fclose(fp); fp = 0; return false;
79  }
80  unsigned int info_size = GET_4B(bmp_info,0);
81  if (!(info_size == 12 || info_size == 40 || info_size == 64)) {
82  last_error = "unknown size of bmp info"; fclose(fp); fp = 0; return false;
83  }
84  if (fread(bmp_info+4, 1, info_size-4, fp) != info_size-4) {
85  last_error = "could not read bmp info"; fclose(fp); fp = 0; return false;
86  }
87  unsigned int width = 0, height = 0, planes = 1, bits_per_pixel = 1;
88  switch ((int) info_size) {
89  case 12: /* Decode OS/2 1.x header (Microsoft calls this a BITMAPCOREHEADER) */
90  width = GET_2B(bmp_info,4);
91  height = GET_2B(bmp_info,6);
92  planes = GET_2B(bmp_info,8);
93  bits_per_pixel = GET_2B(bmp_info,10);
94  break;
95  case 40:
96  case 64:
97  /* Decode Windows 3.x header (Microsoft calls this a BITMAPINFOHEADER) */
98  /* or OS/2 2.x header, which has additional fields that we ignore */
99  width = GET_4B(bmp_info,4);
100  height = GET_4B(bmp_info,8);
101  planes = GET_2B(bmp_info,12);
102  bits_per_pixel = GET_2B(bmp_info,14);
103  /*biXPelsPerMeter = GET_4B(bmp_info,24), biYPelsPerMeter = GET_4B(bmp_info,28), biSizeImage, biClrImportant fields are ignored */
104  if (GET_4B(bmp_info,16) != 0) {
105  last_error = "compressed bmp files not supported"; fclose(fp); fp = 0; return false;
106  }
107  if (bits_per_pixel == 8 && GET_4B(bmp_info,32) != 256) {
108  last_error = "only bmp palettes with 256 entries supported allowed"; fclose(fp); fp = 0; return false;
109  }
110  break;
111  }
112  if (planes != 1) {
113  last_error = "bad number of planes in bmp file"; fclose(fp); fp = 0; return false;
114  }
115  if (!(bits_per_pixel == 8 || bits_per_pixel == 24)) {
116  last_error = "bad number of bits per pixel in bmp file"; fclose(fp); fp = 0; return false;
117  }
118 
119  /* Compute distance to bitmap data --- will adjust for colormap below */
120  int skip_bytes = GET_4B(bmp_header,10) - (info_size + 14);
121 
122  /* Read the colormap, if any */
123  if (bits_per_pixel == 8) {
124  int nr_entries = (info_size == 12) ? 3 : 4;
125  palette.resize(nr_entries*256);
126  if (fread(&palette[0], 1, nr_entries*256, fp) != (size_t)(nr_entries*256)) {
127  last_error = "could not read bmp color map"; palette.clear(); fclose(fp); fp = 0; return false;
128  }
129  skip_bytes -= nr_entries*256;
130  for (unsigned int i = 0; i < 256; ++i) {
131  unsigned char tmp = palette[nr_entries*i];
132  palette[3*i] = palette[nr_entries*i+2];
133  palette[3*i+1] = palette[nr_entries*i+1];
134  palette[3*i+2] = tmp;
135  }
136  }
137  /* Skip any remaining pad bytes */
138  if (skip_bytes < 0) {
139  last_error = "bmp incorrect bfOffBits value?"; palette.clear(); fclose(fp); fp = 0; return false;
140  }
141  while (--skip_bytes >= 0) {
142  fgetc(fp);
143  }
144  if (bits_per_pixel == 8 && palette_formats) {
145  df = data_format(width, height, TI_UINT8, "0");
146  palette_formats->push_back(data_format(256, TI_UINT8, CF_RGB));
147  }
148  else
149  df = data_format(width, height, TI_UINT8, CF_RGB);
150  return true;
151 }

174 bool bmp_reader::read_line(const data_format& df, const data_view& dv)
175 {
176 // static unsigned char temp[4];
177  unsigned char* data_ptr = dv.get_ptr<unsigned char>();
178  unsigned int n = df.get_width();
179  if (palette.empty()) {
180  if (fread(data_ptr, 3, n, fp) != n) {
181  last_error = "bmp read error"; return false;
182  }
183  for (unsigned short x = 0; x < n; ++x, data_ptr +=3)
184  std::swap(data_ptr[0], data_ptr[2]);
185  n *= 3;
186  }
187  else {
188  if (fread(data_ptr, 1, n, fp) != n) {
189  last_error = "bmp read error"; return false;
190  }
191  if (dv.get_format()->get_nr_components() == 3) {
192  data_ptr += n;
193  unsigned char* dest_ptr = data_ptr + 2*n;
194  for (unsigned short x = 0; x < n; ++x) {
195  --data_ptr;
196  dest_ptr -= 3;
197  const unsigned char* p = &palette[3*(*data_ptr)];
198  dest_ptr[0] = p[0];
199  dest_ptr[1] = p[1];
200  dest_ptr[2] = p[2];
201  }
202  }
203  }
204  while ((n & 3) != 0) {
205  fgetc(fp);
206  n++;
207  }
208  return true;
209 }
