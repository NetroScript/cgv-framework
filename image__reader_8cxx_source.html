140 bool image_reader::open(const std::string& file_name)
141 {
142  if (!file_format_ptr)
143  return false;
144  std::string::size_type pos = file_name.find_last_of('.');
145  if (pos == std::string::npos)
146  return false;
147  std::string ext = to_lower(file_name.substr(pos+1));
148  std::vector<base_ptr>& readers = reader_listener::ref_readers();
149  std::stringstream all_supported_extensions("");
150  for (unsigned int i=0; i<readers.size(); ++i) {
151  if (i != readers.size() - 1) {
152  all_supported_extensions << ":";
153  }
154  const std::string &supported_extensions = readers[i]->get_interface<abst_image_reader>()->get_supported_extensions();
155  all_supported_extensions << supported_extensions;
156  if (cgv::utils::is_element(ext, supported_extensions)) {
157  rd = readers[i]->get_interface<abst_image_reader>()->clone();
158  return rd->open(file_name, *file_format_ptr, palette_formats);
159  }
160  }
161  last_error = "could not find a suitable reader for " + file_name + " (supported formats are " +
162  all_supported_extensions.str() + ")";
163  return false;
164 }
