205  fvec<T, 3> view_dir = focus-eye;

208  up_dir.normalize();

213  R(0, 0) = x(0); R(0, 1) = x(1); R(0, 2) = x(2);
214  R(1, 0) = y(0); R(1, 1) = y(1); R(1, 2) = y(2);
215  R(2, 0) = -view_dir(0); R(2, 1) = -view_dir(1); R(2, 2) = -view_dir(2);
216  return R*translate4<T>(-eye);

250  fmat<T, 4, 4> M;
251  M.zeros();
252  T angle = T(0.008726646262)*fovy;
253  T f = cos(angle)/sin(angle);
254  M(0, 0) = f / aspect;
