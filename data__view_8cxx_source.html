167 template <class D, typename P>
168 D data_view_impl<D,P>::permute(const std::string& permutation) const
169 {
170  unsigned int n = (unsigned int) permutation.size();
171  if (n < 2 || n > 4) {
172  std::cerr << "permutation '" << permutation.c_str() << "' has invalid length " << n << std::endl;
173  return D();
174  }
175  unsigned int new_step_sizes[4];
176  unsigned int i;
177  bool used[4] = { false, false, false, false };
178  for (i=0; i<4; ++i)
179  new_step_sizes[i] = step_sizes[i];
180  for (i=0; i<n; ++i) {
181  int idx = permutation[i]-'i';
182  if (idx < 0 || idx > 3) {
183  std::cerr << "invalid permutation entry '" << permutation[i] << "', only 'ijkl' allowed" << std::endl;
184  return D();
185  }
186  used[idx] = true;
187  new_step_sizes[i] = step_sizes[idx];
188  }
189  for (i=0; i<get_dim(); ++i)
190  if (!used[i]) {
191  std::cerr << "invalid permutation of length " << n << " without reference to '" << ('i'+i) << "'" << std::endl;
192  return D();
193  }
194  return D(get_format(), get_ptr<unsigned char>(), get_dim(),
195  new_step_sizes);
196 }

350 bool data_view::compose(data_view& composed_dv, const std::vector<data_view>& dvs)
351 {
352  if(dvs.size() > 0) {
353  const data_format* src_df_ptr = dvs[0].format;
354 
355  unsigned n_dims = src_df_ptr->get_nr_dimensions();
356  const component_format cf = src_df_ptr->get_component_format();
357 
358  data_format* composed_df = new data_format(*src_df_ptr);
359 
360  if(n_dims < 1 || n_dims > 3) {
361  std::cerr << "cannot compose data views with " << n_dims << " dimension" << std::endl;
362  return false;
363  }
364 
365  switch(n_dims) {
366  case 1: composed_df->set_height ((unsigned)dvs.size()); break;
367  case 2: composed_df->set_depth ((unsigned)dvs.size()); break;
368  case 3: composed_df->set_nr_time_steps((unsigned)dvs.size()); break;
369  }
370 
371  if(composed_dv.empty()) {
372  new(&composed_dv) data_view(composed_df);
373  } else {
374  std::cerr << "cannot compose into a non empty data view" << std::endl;
375  return false;
376  }
377 
378  unsigned char* dst_ptr = composed_dv.get_ptr<unsigned char>();
379  unsigned wrong_format_count = 0;
380 
381  size_t bytes_per_slice = composed_df->get_nr_bytes() / dvs.size();
382 
383  for(size_t i = 0; i < dvs.size(); ++i) {
384  const data_view& dv = dvs[i];
385  const data_format* df_ptr = dv.get_format();
386  unsigned char* src_ptr = dv.get_ptr<unsigned char>();
387  size_t n_bytes = df_ptr->get_nr_bytes();
388 
389  if(*src_df_ptr != *df_ptr || n_bytes != bytes_per_slice) {
390  ++wrong_format_count;
391  continue;
392  }
393 
394  memcpy(dst_ptr, src_ptr, n_bytes);
395  dst_ptr += n_bytes;
396  }
397 
398  if(wrong_format_count > 0) {
399  std::cerr << "skipped " << wrong_format_count << " data views with unmatching formats while composing" << std::endl;
400  return false;
401  }
402  return true;
403  }

408 bool data_view::combine_components(data_view& dv, const std::vector<data_view>::iterator first, const std::vector<data_view>::iterator last) {
409 
410  unsigned n_components = (unsigned)std::distance(first, last);
411  if(n_components < 2 || n_components > 4) {
412  std::cerr << "cannot combine channels of less than 2 or more than 4 data views" << std::endl;
413  return false;
414  }
415 
416  const data_format* src_df_ptr = first->format;
417 
418  const component_format src_cf = src_df_ptr->get_component_format();
419  if(src_cf.get_nr_components() > 1) {
420  std::cerr << "cannot combine components of data views with more than one component" << std::endl;
421  return false;
422  }
423 
424  std::vector<data_view>::iterator it = first;
425  for(unsigned i = 1; i < n_components; ++i) {
426  const data_format* df_ptr = it->format;
427 
428  if(*src_df_ptr != *df_ptr) {
429  std::cerr << "cannot combine channels of data views with different formats" << std::endl;
430  return false;
431  }
432  ++it;
433  }
434 
435  unsigned n_dims = src_df_ptr->get_nr_dimensions();
436  cgv::type::info::TypeId component_type = src_df_ptr->get_component_type();
437 
438  ComponentFormat dst_component_format = CF_R;
439  switch(n_components) {
440  case 2: dst_component_format = CF_RG; break;
441  case 3: dst_component_format = CF_RGB; break;
442  case 4: dst_component_format = CF_RGBA; break;
443  }
444 
445  data_format* dst_df_ptr = nullptr;
446  unsigned w, h, d, t;
447  w = h = d = t = 1;
448 
449  unsigned mask_h = 0;
450  unsigned mask_d = 0;
451  unsigned mask_t = 0;
452 
453  switch(n_dims) {
454  case 1:
455  w = src_df_ptr->get_width();
456  dst_df_ptr = new data_format(w, component_type, dst_component_format);
457  break;
458  case 2:
459  w = src_df_ptr->get_width();
460  h = src_df_ptr->get_height();
461  mask_h = 1;
462  dst_df_ptr = new data_format(w, h, component_type, dst_component_format);
463  break;
464  case 3:
465  w = src_df_ptr->get_width();
466  h = src_df_ptr->get_height();
467  d = src_df_ptr->get_depth();
468  mask_h = 1;
469  mask_d = 1;
470  dst_df_ptr = new data_format(w, h, d, component_type, dst_component_format);
471  break;
472  case 4:
473  w = src_df_ptr->get_width();
474  h = src_df_ptr->get_height();
475  d = src_df_ptr->get_depth();
476  t = src_df_ptr->get_nr_time_steps();
477  mask_h = 1;
478  mask_d = 1;
479  mask_t = 1;
480  dst_df_ptr = new data_format(w, h, d, t, component_type, dst_component_format);
481  break;
482  }
483 
484  if(dv.empty()) {
485  new(&dv) data_view(dst_df_ptr);
486  } else {
487  std::cerr << "cannot combine channels into a non empty data view" << std::endl;
488  return false;
489  }
490 
491  //component_type
492  unsigned component_size = cgv::type::info::get_type_size(component_type);
493 
494  for(unsigned i = 0; i < t; ++i) {
495  for(unsigned z = 0; z < d; ++z) {
496  for(unsigned y = 0; y < h; ++y) {
497  for(unsigned x = 0; x < w; ++x) {
498 
499  it = first;
500  for(unsigned j = 0; j < n_components; ++j) {
501  const data_view& src_dv = (*it);
502  unsigned char* src_ptr = src_dv.get_ptr<unsigned char>(x, mask_h*y, mask_d*z, mask_t*i);

504 
505  dst_ptr += j * component_size;
506
