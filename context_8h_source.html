110  RPF_SET_STATE_FLAGS = 1 << 11, // whether to set depth buffer and culling flags
111  RPF_SET_CLEAR_COLOR = 1 << 12, // whether to set the clear color
112  RPF_SET_CLEAR_DEPTH = 1 << 13, // whether to set the clear color
