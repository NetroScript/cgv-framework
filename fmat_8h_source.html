154  const fmat<T,N,N> mul_h (const fmat<S,N-1,N-1>& m2) const
155  {
156  static_assert(N == M, "Number of rows and columns must be equal");
157  static const auto vzero = fvec<T, N-1>(0);
158  fvec<T,N> rows[N]; // extracting a row takes linear time so we only want to do it once for each row
159  for (unsigned i=0; i<N; i++)
160  rows[i] = row(i);
161  fmat<T,N,N> r;
162  // (1) multiply with implied N x (N-1) matrix that is assumed to have an all-zero last row
163  for (unsigned j=0; j<N-1; j++)
164  for (unsigned i=0; i<N; i++)
165  r(i,j) = dot_dir(rows[i], m2.col(j));
166  // (2) assume homogeneous zero position vector in last column of 2nd operand for calculating last result column
167  for (unsigned i=0; i<N; i++)
168  r(i,N-1) = dot_pos(rows[i], vzero);
169  return r;
170  }
