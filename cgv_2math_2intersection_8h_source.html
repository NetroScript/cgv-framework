23 int ray_box_intersection(const ray<T, 3>& ray, fvec<T, 3> extent, fvec<T, 2>& out_ts, fvec<T, 3>* out_normal = nullptr) {
24 
25  fvec<T, 3> m = fvec<T, 3>(T(1)) / ray.direction; // could be precomputed if traversing a set of aligned boxes
26  fvec<T, 3> n = m * ray.origin; // could be precomputed if traversing a set of aligned boxes
27  fvec<T, 3> k = abs(m) * extent;
28  fvec<T, 3> t1 = -n - k;
29  fvec<T, 3> t2 = -n + k;
30  T t_near = std::max(std::max(t1.x(), t1.y()), t1.z());
31  T t_far = std::min(std::min(t2.x(), t2.y()), t2.z());
32 
33  if(t_near > t_far || t_far < T(0))
34  return 0;
35 
36  out_ts[0] = t_near;
37  out_ts[1] = t_far;
38 
39  if(out_normal)
40  *out_normal = -sign(ray.direction)
41  * step(fvec<T, 3>(t1.y(), t1.z(), t1.x()), fvec<T, 3>(t1.x(), t1.y(), t1.z()))
42  * step(fvec<T, 3>(t1.z(), t1.x(), t1.y()), fvec<T, 3>(t1.x(), t1.y(), t1.z()));
43 
44  return 2;
45 }

60  fvec<T, 3> t1 = (max - ray.origin) / ray.direction;
61

63  std::swap(t0.x(), t1.x());
64 
65  if(t0.y() > t1.y())
66  std::swap(t0.y(), t1.y());

69  std::swap(t0.z(), t1.z());
70 
71  if(t0.x() > t1.y() || t0.y() > t1.x() ||
72  t0.x() > t1.z() || t0.z() > t1.x() ||
73  t0.z() > t1.y() || t0.y() > t1.z())

99 template <typename T>
100 int ray_cylinder_intersection(const ray<T, 3>& ray, const fvec<T, 3>& position, const fvec<T, 3>& axis, T radius, T& out_t, fvec<T, 3>* out_normal = nullptr) {
101 
102  fvec<T, 3> oc = ray.origin - position;
103  T caca = dot(axis, axis);
104  T card = dot(axis, ray.direction);
105  T caoc = dot(axis, oc);
106  T a = caca - card * card;
107  T b = caca * dot(oc, ray.direction) - caoc * card;
108  T c = caca * dot(oc, oc) - caoc * caoc - radius * radius * caca;
109  T h = b * b - a * c;
110 
111  if(h < T(0))
112  return 0;
113 
114  h = std::sqrt(h);
115  out_t = (-b - h) / a;
116 
117  // body
118  T y = caoc + out_t * card;
119  if(y > T(0) && y < caca) {
120  if(out_normal)
121  *out_normal = (oc + out_t * ray.direction - axis * y / caca) / radius;
122  return 1;
123  }
124 
125  // caps
126  out_t = ((y < T(0) ? T(0) : caca) - caoc) / card;
127  if(std::abs(b + a * out_t) < h) {
128  if(out_normal)
129  *out_normal = axis * sign(y) / caca;
130  return 1;
131  }
132 
133  return 0;
134 }

177 int ray_sphere_intersection(const ray<T, 3>& ray, const fvec<T, 3>& center, T radius, fvec<T, 2>& out_ts) {
178 
179  fvec<T, 3> d = ray.origin - center;
180  T il = T(1) / dot(ray.direction, ray.direction);
181  T b = il * dot(d, ray.direction);
182  T c = il * (dot(d, d) - radius * radius);
183  T D = b * b - c;
184 
185  if(D < T(0))
186  return 0;
187 
188  if(D < std::numeric_limits<T>::epsilon()) {
189  out_ts = -b;
190  return 1;
191  }
192 
193  D = std::sqrt(D);
194  out_ts[0] = -b - D;
195  out_ts[1] = -b + D;
196 
197  return 2;
198 }

235 template <typename T>
236 int ray_torus_intersection(const ray<T, 3>& ray, T large_radius, T small_radius, T& out_t, fvec<T, 3>* out_normal = nullptr) {
237 
238  T po = T(1);
239  T Ra2 = large_radius * large_radius;
240  T ra2 = small_radius * small_radius;
241  T m = dot(ray.origin, ray.origin);
242  T n = dot(ray.origin, ray.direction);
243  T k = (m + Ra2 - ra2) / T(2);
244  T k3 = n;
245  const fvec<T, 2>& ro_xy = reinterpret_cast<const fvec<T, 2>&>(ray.origin);
246  const fvec<T, 2>& rd_xy = reinterpret_cast<const fvec<T, 2>&>(ray.direction);
247  T k2 = n * n - Ra2 * dot(rd_xy, rd_xy) + k;
248  T k1 = n * k - Ra2 * dot(rd_xy, ro_xy);
249  T k0 = k * k - Ra2 * dot(ro_xy, ro_xy);
250 
251  if(std::abs(k3 * (k3 * k3 - k2) + k1) < T(0.01)) {
252  po = T(-1);
253  T tmp = k1; k1 = k3; k3 = tmp;
254  k0 = T(1) / k0;
255  k1 = k1 * k0;
256  k2 = k2 * k0;
257  k3 = k3 * k0;
258  }
259 
260  T c2 = k2 * T(2) - T(3) * k3 * k3;
261  T c1 = k3 * (k3 * k3 - k2) + k1;
262  T c0 = k3 * (k3 * (c2 + T(2) * k2) - T(8) * k1) + T(4) * k0;
263  c2 /= T(3);
264  c1 *= T(2);
265  c0 /= T(3);
266  T Q = c2 * c2 + c0;
267  T R = c2 * c2 * c2 - T(3) * c2 * c0 + c1 * c1;
268  T h = R * R - Q * Q * Q;
269 
270  if(h >= T(0)) {
271  h = std::sqrt(h);
272  T v = sign(R + h) * std::pow(std::abs(R + h), T(1) / T(3)); // cube root
273  T u = sign(R - h) * std::pow(std::abs(R - h), T(1) / T(3)); // cube root
274  fvec<T, 2> s = fvec<T, 2>((v + u) + T(4) * c2, (v - u) * std::sqrt(T(3)));
275  T y = std::sqrt(T(0.5) * (length(s) + s.x()));
276  T x = T(0.5) * s.y() / y;
277  T r = T(2) * c1 / (x * x + y * y);
278  T t1 = x - r - k3; t1 = (po < T(0)) ? T(2) / t1 : t1;
279  T t2 = -x - r - k3; t2 = (po < T(0)) ? T(2) / t2 : t2;
280 
281  if(t1 > T(0)) out_t = t1;
282  if(t2 > T(0)) out_t = std::min(out_t, t2);
283 
284  if(out_normal) {
285  fvec<T, 3> pos = ray.position(out_t);
286  *out_normal = normalize(pos * ((dot(pos, pos) - ra2) * fvec<T, 3>(T(1)) - Ra2 * fvec<T, 3>(T(1), T(1), T(-1))));
287  }
288 
289  return 1; // 2
290  }
291 
292  T sQ = std::sqrt(Q);
293  T w = sQ * cos(acos(-R / (sQ * Q)) / T(3));
294  T d2 = -(w + c2);
295 
296  if (d2 < T(0))
297  return 0;
298 
299  T d1 = std::sqrt(d2);
300  T h1 = std::sqrt(w - T(2) * c2 + c1 / d1);
301  T h2 = std::sqrt(w - T(2) * c2 - c1 / d1);
302  T t1 = -d1 - h1 - k3; t1 = (po < T(0)) ? T(2) / t1 : t1;
303  T t2 = -d1 + h1 - k3; t2 = (po < T(0)) ? T(2) / t2 : t2;
304  T t3 = d1 - h2 - k3; t3 = (po < T(0)) ? T(2) / t3 : t3;
305  T t4 = d1 + h2 - k3; t4 = (po < T(0)) ? T(2) / t4 : t4;
306 
307  if (t1 > T(0)) out_t = t1;
308  if (t2 > T(0)) out_t = std::min(out_t, t2);
309  if (t3 > T(0)) out_t = std::min(out_t, t3);
310  if (t4 > T(0)) out_t = std::min(out_t, t4);
311 
312  if(out_normal) {
313  fvec<T, 3> pos = ray.position(out_t);
314  *out_normal = normalize(pos * ((dot(pos, pos) - ra2) * fvec<T, 3>(T(1)) - Ra2 * fvec<T, 3>(T(1), T(1), T(-1))));
315  }
316 
317  return 1; // 4
318 }

330 template <typename T>
331 int ray_torus_intersection(const ray<T, 3>& ray, const fvec<T, 3>& center, const fvec<T, 3>& normal, T large_radius, T small_radius, T& out_t, fvec<T, 3>* out_normal = nullptr) {
332 
333  // compute pose transformation
334  fmat<T, 3, 4> pose;
335  cgv::math::pose_position(pose) = center;
336  fvec<T, 3>& x = reinterpret_cast<fvec<T, 3>&>(pose[0]);
337  fvec<T, 3>& y = reinterpret_cast<fvec<T, 3>&>(pose[3]);
338  fvec<T, 3>& z = reinterpret_cast<fvec<T, 3>&>(pose[6]);
339  z = normal;
340  x = normal;
341  int i = std::abs(normal[0]) < std::abs(normal[1]) ? 0 : 1;
342  i = std::abs(normal[i]) < std::abs(normal[2]) ? i : 2;
343  x[i] = T(1);
344  y = normalize(cross(normal, x));
345  x = cross(y, normal);
346 
347  cgv::math::ray<T, 3> transformed_ray;
348  transformed_ray.origin = cgv::math::inverse_pose_transform_point(pose, ray.origin);
349  transformed_ray.direction = cgv::math::inverse_pose_transform_vector(pose, ray.direction);
350 
351  // transform ray into torus pose
352  int res = ray_torus_intersection(transformed_ray, large_radius, small_radius, out_t, out_normal);
353 
354  // in case of intersection, transform normal back to world space
355  if(res)
356  *out_normal = cgv::math::pose_transform_vector(pose, *out_normal);
357 
358  return res;
359 }
