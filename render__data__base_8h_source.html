212  virtual bool transfer(context& ctx, renderer& r) {
213  state_out_of_date = false;
214 
215  // The positions array determines the amount of stored vertices.
216  // All other arrays except indices must match its size.
217  if(!positions.empty()) {
218  r.set_position_array(ctx, positions);
219  if(colors.size() == size())
220  r.set_color_array(ctx, colors);
221  if(!indices.empty())
222  r.set_indices(ctx, indices);
223  else
224  r.remove_indices(ctx);
225  return true;
226  } else {
227  // If no positions are set the render data is assumed to be empty. In this case the
228  // attribute array manager is destructed and reinitialized to clear the GPU buffers.
229  if(attribute_array.is_created()) {
230  attribute_array.destruct(ctx);
231  attribute_array.init(ctx);
232  }
233  // Return false since the attribute array manager holds no data anymore.
234  return false;
235  }
236  }
