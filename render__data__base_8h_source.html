216  virtual bool transfer(context& ctx, renderer& r) {
217  state_out_of_date = false;
218 
219  // The positions array determines the amount of stored vertices.
220  // All other arrays except indices must match its size.
221  if(!positions.empty()) {
222  r.set_position_array(ctx, positions);
223  if(colors.size() == size())
224  r.set_color_array(ctx, colors);
225  if(!indices.empty())
226  r.set_indices(ctx, indices);
227  else
228  r.remove_indices(ctx);
229  return true;
230  } else {
231  // If no positions are set the render data is assumed to be empty. In this case the
232  // attribute array manager is destructed and reinitialized to clear the GPU buffers.
233  if(attribute_array.is_created()) {
234  attribute_array.destruct(ctx);
235  attribute_array.init(ctx);
236  }
237  // Return false since the attribute array manager holds no data anymore.
238  return false;
239  }
240  }
