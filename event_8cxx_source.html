59 unsigned char stream_in_toggle_keys(std::istream& is)
60 {
61  int c = is.peek();
62  if (c > 255)
63  return 0;
64  c = cgv::utils::to_upper(c);
65  int ti;
66  for (ti = 0; ti < nr_togs; ++ti)
67  if (tog_names[ti][0] == c)
68  break;
69  if (ti == nr_togs)
70  return 0;
71 
72  int n = (int)std::string(tog_names[ti]).size();
73  int buffer[MAX_TOG_LENGTH];
74  for (int i=1; i<n; ++i) {
75  buffer[i-1] = is.get();
76  c = is.peek();
77  if (c > 255)
78  return 0;
79  c = cgv::utils::to_lower(c);
80  if (tog_names[ti][i] != c) {
81  for (int j = i; j > 0; ) {
82  --j;
83  is.putback(buffer[j]);
84  }
85  return 0;
86  }
87  }
88  is.get();
89  return tog_ids[ti];
90 }
