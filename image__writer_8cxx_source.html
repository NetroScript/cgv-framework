48 image_writer::image_writer(const std::string& _file_name) : file_name(_file_name)
49 {
50  is_opened = false;
51  std::string::size_type pos = file_name.find_last_of('.');
52  if (pos == std::string::npos) {
53  last_error = "file name without extension specified";
54  return;
55  }
56  std::string ext = to_lower(file_name.substr(pos+1));
57  std::vector<base_ptr>& writers = writer_listener::ref_writers();
58  for (unsigned int i=0; i<writers.size(); ++i) {
59  abst_image_writer* aiw = writers[i]->get_interface<abst_image_writer>();
60  if (cgv::utils::is_element(ext, aiw->get_supported_extensions())) {
61  wr = aiw->clone();
62  return;
63  }
64  }
65  wr = 0;
66  last_error = "no image image_writer found for extension: ";
67  last_error += ext;
68 }
