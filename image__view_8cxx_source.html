26  for (size_t j=0; j<W;++j) {
27  for (size_t k=0; k<dx; ++k) {
28  *q = (T) ( ( (S)p0[0] + p1[0] + p0[dx] + p1[dx] ) / 4 );
29  ++q;
30  ++p0;
31  ++p1;
32  }
33  p0 += dx;
34  p1 += dx;
35  }
36  p0 += 2*dy - 2*W*dx;
37  p1 += 2*dy - 2*W*dx;
38  q += Dy - W*Dx;
39  }
