74  void volume_renderer::init_noise_texture(context& ctx)
75  {
76  if(noise_texture.is_created())
77  noise_texture.destruct(ctx);
78 
79  unsigned size = 64;
80  std::vector<float> noise_data(size*size);
81 
82  std::default_random_engine rng(42);
83  std::uniform_real_distribution<float> dist(0.0f, 1.0f);
84 
85  for(size_t i = 0; i < noise_data.size(); ++i)
86  noise_data[i] = dist(rng);
87 
88  cgv::data::data_view dv = cgv::data::data_view(new cgv::data::data_format(size, size, TI_FLT32, cgv::data::CF_R), noise_data.data());
89  noise_texture.create(ctx, dv, 0);
90  }

199  bool volume_renderer::enable(context& ctx)
200  {
201  const volume_render_style& vrs = get_style<volume_render_style>();
202  if (!renderer::enable(ctx))
203  return false;
204  int vp[4];
205  glGetIntegerv(GL_VIEWPORT, vp);
206  ref_prog().set_uniform(ctx, "viewport_dims", vec2(float(vp[2]-vp[0]), float(vp[3]-vp[1])));
207  ref_prog().set_uniform(ctx, "noise_offset", noise_offset);
208 
209  ref_prog().set_uniform(ctx, "opacity_scale", vrs.opacity_scale);
210  ref_prog().set_uniform(ctx, "size_scale", vrs.size_scale);
211 
212  ref_prog().set_uniform(ctx, "light_local_to_eye", vrs.light_local_to_eye);
213  ref_prog().set_uniform(ctx, "light_direction", normalize(vrs.light_direction));
214  float specular_power = cgv::math::lerp(128.0f, 1.0f, cgv::math::clamp(vrs.roughness, 0.0f, 1.0f));
215  ref_prog().set_uniform(ctx, "light_parameters", vec4(vrs.ambient_strength, vrs.diffuse_strength, vrs.specular_strength, specular_power));
216 
217  ref_prog().set_uniform(ctx, "gradient_lambda", std::max(vrs.gradient_lambda, 0.001f));
218 
219  ref_prog().set_uniform(ctx, "isovalue", vrs.isovalue);
220  ref_prog().set_uniform(ctx, "isosurface_color", vrs.isosurface_color);
221 
222  ref_prog().set_uniform(ctx, "clip_box_min", vrs.clip_box.get_min_pnt());
223  ref_prog().set_uniform(ctx, "clip_box_max", vrs.clip_box.get_max_pnt());
224 
225  glDisable(GL_DEPTH_TEST);
226  glEnable(GL_BLEND);
227  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
228  glCullFace(GL_FRONT);
229  glEnable(GL_CULL_FACE);
230 
231  if(volume_texture) volume_texture->enable(ctx, 0);
232  if(transfer_function_texture) transfer_function_texture->enable(ctx, 1);
233  noise_texture.enable(ctx, 2);
234  if(gradient_texture) gradient_texture->enable(ctx, 3);
235  if(depth_texture) depth_texture->enable(ctx, 4);
236  return true;
237  }
